1 /**
  2 * Class that provides the UI for the dapp application. Reads the JSON file containing
  3 * the information of the events and load in the page.
  4 **/
  5 App = {
  6   web3Provider: null,
  7   contracts: {},
  8 
  9   /**
 10   * Initialization function that starts the provider from web3 and load the content
 11   * of JSON file.
 12   **/
 13   init: function() {
 14 
 15     $.getJSON('../pets.json', function(data) {
 16       var eventsRow = $('#eventsRow');
 17       var eventTemplate = $('#eventTemplate');
 18 
 19       for (i = 0; i < data.length; i ++) {
 20         eventTemplate.find('.panel-title').text(data[i].name);
 21         eventTemplate.find('img').attr('src', data[i].picture);
 22         eventTemplate.find('.pet-location').text(data[i].location);
 23         eventTemplate.find('.btn-sub').attr('data-id', data[i].id);
 24         eventsRow.append(eventTemplate.html());
 25       }
 26     });
 27 
 28     return App.initWeb3();
 29   },
 30 
 31   /**
 32   * Function that initalizes the web3 provider. Web3 is a plugin/framework
 33   * that handles the transactions on the block chain.
 34   * @constructor
 35   **/
 36   initWeb3: function() {
 37 
 38     if (typeof web3 !== 'undefined') {
 39       App.web3Provider = web3.currentProvider;
 40     }
 41     else {
 42       App.web3Provider = new Web3.providers.HttpProvider('http://localhost:7545');
 43     }
 44     web3 = new Web3(App.web3Provider);
 45     return App.initContract();
 46   },
 47 
 48   /**
 49   * Function the initalizes the contract and link with the web provider. Use
 50   * the TruffleContract library to get an instance of the contract.
 51   * @constructor
 52   */
 53   initContract: function() {
 54     $.getJSON('Adoption.json', function(data) {
 55         var EventArtifact = data;
 56         App.contracts.Adoption = TruffleContract(EventArtifact);
 57         App.contracts.Adoption.setProvider(App.web3Provider);
 58     });
 59     return App.bindEvents();
 60   },
 61 
 62   /**
 63   * Function to watch events on te button click and call the right functions.
 64   **/
 65   bindEvents: function() {
 66     $(document).on('click', '.btn-sub', App.subscribe);
 67     $(document).on('click', '.btn-unsub', App.unscribe);
 68   },
 69 
 70   /**
 71   * Subscribe function responsible to save that one account registered for the event.
 72   * The transaction confirms the registration belongs to the contract.
 73   **/
 74   subscribe: function(event) {
 75     event.preventDefault();
 76 
 77     var eventId = parseInt($(event.target).data('id'));
 78     var quota = 10;
 79     var eventInstance;
 80 
 81 web3.eth.getAccounts(function(error, accounts) {
 82   if (error) {
 83     console.log(error);
 84   }
 85 
 86   var account = accounts[0];
 87 
 88   App.contracts.Adoption.deployed().then(function(instance) {
 89     eventInstance = instance;
 90     return eventInstance.subscribe(eventId, quota, {from: account, value: 1, gas: 1000000});
 91   }).then(function(result) {
 92     console.log(result);
 93   });
 94 });
 95 },
 96 
 97   /**
 98   * Function that removes the subscription of the contract in the event.
 99   *
100   */
101   unscribe: function(event) {
102     event.preventDefault();
103 
104     var eventId = parseInt($(event.target).data('id'));
105 
106     var eventInstance;
107 
108 web3.eth.getAccounts(function(error, accounts) {
109   if (error) {
110     console.log(error);
111   }
112 
113   var account = accounts[0];
114 
115   App.contracts.Adoption.deployed().then(function(instance) {
116     eventInstance = instance;
117 
118     return eventInstance.unscribe(eventId, {from: account, value: 1, gas: 1000000});
119   }).then(function(result) {
120     console.log('Unregistered');
121   }).catch(function(err) {
122     console.log(err.message);
123   });
124 });
125   }
126 
127 };
128 
129 /**
130 * Initialization function.
131 */
132 $(function() {
133   $(window).load(function() {
134     App.init();
135   });
136 });
